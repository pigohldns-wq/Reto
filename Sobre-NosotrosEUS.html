01. Lerro-jarraitzaile Robota

Arduino UNO

Robot autonomo honek hondo zuriaren gaineko lerro beltzak detektatzen ditu infragorri-sentsoreen bidez. Bere logika serbomotoreen etengabeko doikuntzan oinarritzen da ibilbideari eusteko.

#include 

Servo motorIzquierdo; 
Servo motorDerecho;

int sensorIzq = 2;
int sensorDer = 3;

// --- ABIADURA BALIO EGOKITUAK ---
// 90 stop da. 
// Ezkerreko motorrarentzat: 0 aurrera maximoa da, 180 atzera.
// Eskuinreko motorrarentzat: 180 aurrera maximoa da, 0 atzera.

int velRapida_Iz = 50;   // Ezkerreko gurutze-abiadura
int velRapida_Der = 130;  // Eskuinreko gurutze-abiadura

void setup() {
  motorIzquierdo.attach(8);
  motorDerecho.attach(4);
  pinMode(sensorIzq, INPUT);
  pinMode(sensorDer, INPUT);

  // Segurtasun geldialdia hasieran
  motorIzquierdo.write(90);
  motorDerecho.write(90);
}

void loop() {
  int lecturaIzq = digitalRead(sensorIzq);
  int lecturaDer = digitalRead(sensorDer);

  // --- LOGIKA: BELTZA = SEGURUA (Mugitu) | ZURIA = EZ-SEGURUA (Zuzendu) ---

  if (lecturaIzq == LOW && lecturaDer == LOW) {
    // BIak BELTZEAN: Aurrera!
    motorIzquierdo.write(velRapida_Iz);
    motorDerecho.write(velRapida_Der); 
  } 
  else if (lecturaIzq == HIGH && lecturaDer == LOW) {
    // EZKERRA ZURIAN: Beltza eskuinean dago. 
    // Eskuinera poliki biratu: ezkerreko gurpila mugitu eta eskuinekoa gelditu.
    motorIzquierdo.write(velRapida_Iz); 
    motorDerecho.write(90); 
  } 
  else if (lecturaIzq == LOW && lecturaDer == HIGH) {
    // ESKUINA ZURIAN: Beltza ezkerrean dago.
    // Ezkerrera poliki biratu: ezkerrekoa gelditu eta eskuinekoa mugitu.
    motorIzquierdo.write(90); 
    motorDerecho.write(velRapida_Der);
  }
  else {
    // BIAK ZURIAN: Erabat irten da.
    // Beltza bilatzeko erreskate-biraketa txikia.
    motorIzquierdo.write(70); 
    motorDerecho.write(70); 
  }
}

#include 

Servo motor4; // ESKUINA - Pin 4
Servo motor8; // EZKERRA - Pin 8

void setup() { 
  motor4.attach(4);
  motor8.attach(8);
  
  // Hasierako 2 segundoko geldialdia askatzeko denbora emateko
  motor4.write(90);
  motor8.write(90);
  delay(2000);
}

void loop() {
  // --- 1. ESKUINERA biratu ---
  // (Ezkerra mugitu (8), Eskuina gelditu (4))
  motor4.write(90);   // Pin 4: GELDITUTA
  motor8.write(0);    // Pin 8: AURRERA (Lehengo -180 ordezkatzen du)
  delay(400);

  // --- 2. ZUZEN joan ---
  motor4.write(180);  // Pin 4: AURRERA
  motor8.write(0);    // Pin 8: AURRERA
  delay(6500);

  // --- 3. EZKERRERA biratu ---
  // (Ezkerra gelditu (8), Eskuina mugitu (4))
  motor4.write(180);  // Pin 4: AURRERA
  motor8.write(90);   // Pin 8: GELDITUTA
  delay(850);

  // --- 4. ZUZEN joan ---
  motor4.write(180);
  motor8.write(0);
  delay(7000);

  // --- 5. ESKUINERA biratu ---
  motor4.write(90);
  motor8.write(0);
  delay(1050);

  // --- 6. ZUZEN joan ---
  motor4.write(180);
  motor8.write(0);
  delay(8500);

  // --- 7. EZKERRERA biratu ---
  motor4.write(180);

  motor4.write(90);
  motor8.write(90);
  while(true); 
}

void receiveLED(){

    //L 000000001000010100100011000000000

    unsigned long int matrix;
    char *arg;
    char *endstr;
    arg=SerialCmd.next();
    //Serial.println (arg);

      matrix=strtoul(arg,&endstr,2);    // Karaktere kate bat zeinu gabeko osoko luze bihurtzen du

      robot.clearMouth();
    }

    sendFinalAck();

}
