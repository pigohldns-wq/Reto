01. Robot Sigue-Líneas

Arduino UNO

Un robot autónomo que detecta líneas negras sobre un fondo blanco mediante sensores infrarrojos. Su lógica se basa en el ajuste constante de servomotores para mantener la trayectoria.

#include 

Servo motorIzquierdo; 
Servo motorDerecho;

int sensorIzq = 2;
int sensorDer = 3;

// --- VALORES DE VELOCIDAD AJUSTADOS ---
// 90 es stop. 
// Para el motor izquierdo: 0 es máximo adelante, 180 atrás.
// Para el motor derecho: 180 es máximo adelante, 0 atrás.

int velRapida_Iz = 50;   // Velocidad crucero Izq
int velRapida_Der = 130;  // Velocidad crucero Der

void setup() {
  motorIzquierdo.attach(8);
  motorDerecho.attach(4);
  pinMode(sensorIzq, INPUT);
  pinMode(sensorDer, INPUT);

  // Parada de seguridad al empezar
  motorIzquierdo.write(90);
  motorDerecho.write(90);
}

void loop() {
  int lecturaIzq = digitalRead(sensorIzq);
  int lecturaDer = digitalRead(sensorDer);

  // --- LÓGICA: NEGRO = SEGURO (Moverse) | BLANCO = INSEGURO (Corregir) ---

  if (lecturaIzq == LOW && lecturaDer == LOW) {
    // AMBOS EN NEGRO: ¡Gas!
    motorIzquierdo.write(velRapida_Iz);
    motorDerecho.write(velRapida_Der); 
  } 
  else if (lecturaIzq == HIGH && lecturaDer == LOW) {
    // IZQUIERDA EN BLANCO: El negro está a la derecha. 
    // Giramos suave a la derecha: movemos la rueda izq y paramos la der.
    motorIzquierdo.write(velRapida_Iz); 
    motorDerecho.write(90); 
  } 
  else if (lecturaIzq == LOW && lecturaDer == HIGH) {
    // DERECHA EN BLANCO: El negro está a la izquierda.
    // Giramos suave a la izquierda: paramos la izq y movemos la der.
    motorIzquierdo.write(90); 
    motorDerecho.write(velRapida_Der);
  }
  else {
    // AMBOS EN BLANCO: Se ha salido del todo.
    // Pequeño giro de rescate para buscar el negro.
    motorIzquierdo.write(70); 
    motorDerecho.write(70); 
  }
}

02. Esquiva Obstáculos

Ruta Programada

Navega por el circuito utilizando una ruta programada con tiempos precisos.

#include 

Servo motor4; // DERECHA - Pin 4
Servo motor8; // IZQUIERDA - Pin 8

void setup() { 
  motor4.attach(4);
  motor8.attach(8);
  
  // Parada inicial de 2 segundos para que te dé tiempo a soltarlo
  motor4.write(90);
  motor8.write(90);
  delay(2000);
}

void loop() {
  // --- 1. Girar a la DERECHA ---
  // (Mueve Izquierda (8), para Derecha (4))
  motor4.write(90);   // Pin 4: PARADO
  motor8.write(0);    // Pin 8: ADELANTE (Sustituye al -180)
  delay(400);

  // --- 2. Ir RECTO ---
  motor4.write(180);  // Pin 4: ADELANTE
  motor8.write(0);    // Pin 8: ADELANTE
  delay(6500);

  // --- 3. Girar a la IZQUIERDA ---
  // (Para Izquierda (8), mueve Derecha (4))
  motor4.write(180);  // Pin 4: ADELANTE
  motor8.write(90);   // Pin 8: PARADO
  delay(850);

  // --- 4. Ir RECTO ---
  motor4.write(180);
  motor8.write(0);
  delay(7000);

  // --- 5. Girar a la DERECHA ---
  motor4.write(90);
  motor8.write(0);
  delay(1050);

  // --- 6. Ir RECTO ---
  motor4.write(180);
  motor8.write(0);
  delay(8500);

  // --- 7. Girar a la IZQUIERDA ---
  motor4.write(180);
  motor8.write(90);
  delay(900);

  // --- 8. Ir RECTO ---
  motor4.write(180);
  motor8.write(0);
  delay(6000);

  // --- 9. Girar a la DERECHA ---
  motor4.write(90);
  motor8.write(0);
  delay(500);

  // --- 10. Ir RECTO ---
  motor4.write(180);
  motor8.write(0);
  delay(5000);

  // Al finalizar la ruta, se detiene para no repetir en bucle loco
  motor4.write(90);
  motor8.write(90);
  while(true); 
}

//-- Función para enviar comando Ack (A)
void sendAck(){

  delay(30);

  Serial.print(F("&&"));
  Serial.print(F("A"));
  Serial.println(F("%%"));
  Serial.flush();
}


//-- Función para enviar comando Final Ack (F)
void sendFinalAck(){

  delay(30);

  Serial.print(F("&&"));
  Serial.print(F("F"));
  Serial.println(F("%%"));
  Serial.flush();
}
//-- Función para recibir comandos de movimiento
void receiveMovement(){
    sendAck();
    
    if (robot.getRestState()==true){
        robot.setRestState(false);
    }

    robot.clearMouth();
    //Definición de comandos Bluetooth de Movimiento
    //M  MoveID  T   MoveSize  
    char *arg; 
    arg = SerialCmd.next(); 
    if (arg != NULL) {
      moveId=atoi(arg);
      robot.putMouth(smile);
      }
    else{
      robot.putMouth(xMouth); // mal comando
      delay(2000);
      robot.clearMouth();
      moveId=0; //parar
    }
    
    arg = SerialCmd.next(); 
    if (arg != NULL) {T=atoi(arg);}
    else{
      T=1000;
    }

    arg = SerialCmd.next(); 
    if (arg != NULL) {moveSize=atoi(arg);}
    else{
      moveSize =15;
    }
}
//-- Función para recibir comandos LED
void receiveLED(){  

    //enviar Ack y parar si es necesario
    sendAck();
    robot.home();

    //Ejemplos de comandos Bluetooth receiveLED
    //L 000000001000010100100011000000000
    //L 001111111111111111111111111111111 (todos los LED encendidos)
    unsigned long int matrix;
    char *arg;
    char *endstr;
    arg=SerialCmd.next();
    //Serial.println (arg);
    if (arg != NULL) {
      matrix=strtoul(arg,&endstr,2);    // Convierte una cadena de caracteres a un entero largo sin signo
      robot.putMouth(matrix,false);
    }else{
      robot.putMouth(xMouth); // mal comando
      delay(2000);
      robot.clearMouth();
    }

    sendFinalAck();

}

      BT.println(">> BRAZO IZQ ARRIBA");
      break;

    case 'x':
      brazoIzq.write(180);
      Serial.println(">> BRAZO IZQ ABAJO");
      BT.println(">> BRAZO IZQ ABAJO");
      break;

    case 'l':
      cabeza.write(0);
      Serial.println(">> CABEZA DERECHA");
      BT.println(">> CABEZA DERECHA");
      break;

    case 'j':
      cabeza.write(180);
      Serial.println(">> CABEZA IZQUIERDA");
      BT.println(">> CABEZA IZQUIERDA");
      break;

    case 'k':
      cabeza.write(70);
      Serial.println(">> CABEZA RECTO");
      BT.println(">> CABEZA RECTO");
      break;

    case 'p':
      caderaIzq.write(0);
      caderaDer.write(180);
      Serial.println(">> CADERAS MARCHA");
      BT.println(">> CADERAS MARCHA");
      break;

    case 'o':
      caderaIzq.write(100);
      caderaDer.write(100);
      Serial.println(">> CADERAS NEUTRO");
      BT.println(">> CADERAS NEUTRO");
      break;

    case ' ':
      pararPies();
      Serial.println(">> PARADO");
      BT.println(">> PARADO");
      break;
  }
}
